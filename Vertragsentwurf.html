const heartBg = document.getElementById('heartBg');

for (let i = 0; i < 18; i++) {

  const h = document.createElement('span');

  h.textContent = heartChars[Math.floor(Math.random() * heartChars.length)];

  h.style.left = Math.random() * 100 + '%';

  h.style.fontSize = (14 + Math.random() * 18) + 'px';

  h.style.animationDelay = (Math.random() * 10) + 's';

  heartBg.appendChild(h);

let groesse = 100;

let neinKlicks = 0;

const FLUCHT_GRENZE = 3;


function nein() {

  const btn = document.getElementById('nein-btn');

  neinKlicks++;


  if (btn.style.position !== 'fixed') {

    const rect = btn.getBoundingClientRect();

    btn.style.position = 'fixed';

    btn.style.left = rect.left + 'px';

    btn.style.top = rect.top + 'px';

    btn.style.margin = '0';

    btn.classList.add('running');

  }


  groesse += 25;

  document.getElementById("sticker1").style.width = groesse + "px";


  const rect = btn.getBoundingClientRect();


  if (neinKlicks >= FLUCHT_GRENZE) {

    btn.classList.remove('running');

    btn.classList.add('escaped');

    const richtung = Math.random() < 0.5 ? -1 : 1;

    const zielX = richtung === 1 ? (window.innerWidth + 250) : (-rect.width - 250);

    btn.style.left = zielX + 'px';

    btn.style.top = (Math.random() * (window.innerHeight - rect.height)) + 'px';

    btn.style.transform = 'rotate(' + (richtung * 25) + 'deg)';

    btn.style.opacity = '0';

    btn.textContent = "Nein 🏃💨";


    setTimeout(() => {

      btn.style.display = 'none';

      document.getElementById('escapeNote').style.display = 'block';

    }, 600);

    return;

  }


  const breite = rect.width;

  const hoehe = rect.height;

  const maxX = Math.max(0, window.innerWidth - breite - 20);

  const maxY = Math.max(0, window.innerHeight - hoehe - 20);


  let neuX = Math.random() * maxX;

  let neuY = Math.random() * maxY;


  if (Math.abs(neuX - rect.left) < window.innerWidth * 0.25) {

    neuX = (rect.left + window.innerWidth * 0.5) % maxX;

  }


  btn.style.left = neuX + 'px';

  btn.style.top = neuY + 'px';

  btn.textContent = "Nein 🏃";

  setTimeout(() => { btn.textContent = "Nein"; }, 400);

}


function ja() {

  document.getElementById("sticker1").style.display = "none";

  document.getElementById("sticker2").style.display = "block";

  document.getElementById("jaText").style.display = "block";

}
